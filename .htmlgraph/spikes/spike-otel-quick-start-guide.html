"""Cost analysis queries for delegations.""" from dataclasses import dataclass from typing import List, Optional from datetime import datetime import sqlite3 @dataclass class DelegationMetric: """Single delegation with cost metrics.""" event_id: str timestamp: str subagent_type: str duration_seconds: float estimated_cost_tokens: int child_spike_count: int status: str @property def cost_per_spike(self) -> float: """Cost per spike created.""" if self.child_spike_count == 0: return 0.0 return (self.estimated_cost_tokens * 0.00001) / self.child_spike_count @dataclass class CostSummary: """Aggregated cost metrics.""" total_delegations: int total_cost_tokens: int total_cost_usd: float average_cost: float most_expensive: DelegationMetric least_expensive: DelegationMetric cost_by_subagent: dict delegations: List[DelegationMetric] class CostAnalyzer: """Analyze delegation costs.""" def __init__(self, db_connection): self.db = db_connection def get_delegations( self, session_id: Optional[str] = None, limit: int = 100 ) -> List[DelegationMetric]: """Get all task_delegation events with costs.""" cursor = self.db.cursor() if session_id: cursor.execute(""" SELECT event_id, timestamp, subagent_type, execution_duration_seconds, cost_tokens, child_spike_count, status FROM agent_events WHERE event_type = 'task_delegation' AND session_id = ? ORDER BY timestamp DESC LIMIT ? """, (session_id, limit)) else: cursor.execute(""" SELECT event_id, timestamp, subagent_type, execution_duration_seconds, cost_tokens, child_spike_count, status FROM agent_events WHERE event_type = 'task_delegation' ORDER BY timestamp DESC LIMIT ? """, (limit,)) rows = cursor.fetchall() return [ DelegationMetric( event_id=row[0], timestamp=row[1], subagent_type=row[2], duration_seconds=row[3] or 0.0, estimated_cost_tokens=row[4] or 0, child_spike_count=row[5] or 0, status=row[6] or 'unknown' ) for row in rows ] def get_summary(self, session_id: Optional[str] = None) -> CostSummary: """Get aggregated cost summary.""" delegations = self.get_delegations(session_id) if not delegations: raise ValueError("No delegations found") total_cost_tokens = sum(d.estimated_cost_tokens for d in delegations) total_cost_usd = total_cost_tokens * 0.00001 # Rough estimate # Cost by subagent type cost_by_subagent = {} for delegation in delegations: subagent = delegation.subagent_type or 'unknown' if subagent not in cost_by_subagent: cost_by_subagent[subagent] = { 'count': 0, 'total_cost': 0, 'total_spikes': 0 } cost_by_subagent[subagent]['count'] += 1 cost_by_subagent[subagent]['total_cost'] += delegation.estimated_cost_tokens cost_by_subagent[subagent]['total_spikes'] += delegation.child_spike_count return CostSummary( total_delegations=len(delegations), total_cost_tokens=total_cost_tokens, total_cost_usd=total_cost_usd, average_cost=total_cost_usd / len(delegations), most_expensive=max( delegations, key=lambda d: d.estimated_cost_tokens ), least_expensive=min( delegations, key=lambda d: d.estimated_cost_tokens ), cost_by_subagent=cost_by_subagent, delegations=delegations )

    """Generate HTML cost reports.""" from htmlgraph.analytics.cost_analysis import CostAnalyzer, CostSummary import json class CostReporter: """Generate HTML reports from cost analysis.""" def __init__(self, analyzer: CostAnalyzer): self.analyzer = analyzer def generate_html(self, session_id: str) -> str: """Generate HTML dashboard.""" try: summary = self.analyzer.get_summary(session_id) except ValueError: return self._error_html("No delegations found in this session") # Convert for JSON embedding delegations_json = json.dumps([ { 'event_id': d.event_id, 'subagent': d.subagent_type, 'cost': d.estimated_cost_tokens, 'cost_usd': d.estimated_cost_tokens * 0.00001, 'spikes': d.child_spike_count, 'cost_per_spike': d.cost_per_spike, 'duration': d.duration_seconds } for d in summary.delegations ]) cost_by_type = json.dumps({ k: { 'count': v['count'], 'total_cost': v['total_cost'], 'avg_cost': v['total_cost'] / v['count'] } for k, v in summary.cost_by_subagent.items() }) html = f""" Cost Attribution Dashboard

    Cost Attribution Dashboard

    Session: {session_id}

    {summary.total_delegations}
    Total Delegations
    ${summary.total_cost_usd:.3f}
    Total Cost (est.)
    ${summary.average_cost:.4f}
    Avg Cost/Delegation

    Delegations by Type

    """ max_cost = max(v['total_cost'] for v in summary.cost_by_subagent.values()) for subagent, metrics in summary.cost_by_subagent.items(): avg_cost = metrics['total_cost'] / metrics['count'] cost_usd = metrics['total_cost'] * 0.00001 avg_cost_usd = cost_usd / metrics['count'] # Bar width proportional to cost bar_width = (metrics['total_cost'] / max_cost) * 200 html += f""" """ html += """
    Subagent Type Count Total Cost Avg Cost Visualization
    {subagent} {metrics['count']} ${cost_usd:.4f} ${avg_cost_usd:.4f}

    Most Expensive Delegations

    """ # Show top 10 most expensive sorted_delegations = sorted( summary.delegations, key=lambda d: d.estimated_cost_tokens, reverse=True )[:10] for delegation in sorted_delegations: cost_usd = delegation.estimated_cost_tokens * 0.00001 html += f""" """ html += """
    Type Cost (Tokens) Cost (USD) Duration Spikes Created Cost/Spike
    {delegation.subagent_type} {delegation.estimated_cost_tokens} ${cost_usd:.4f} {delegation.duration_seconds:.1f}s {delegation.child_spike_count} ${delegation.cost_per_spike:.4f}

    Summary Statistics

      """ html += f"""
    • Total delegations: {summary.total_delegations}
    • Most expensive: {summary.most_expensive.subagent_type} (${summary.most_expensive.estimated_cost_tokens * 0.00001:.4f})
    • Least expensive: {summary.least_expensive.subagent_type} (${summary.least_expensive.estimated_cost_tokens * 0.00001:.4f})
    • Total spikes created: {sum(d.child_spike_count for d in summary.delegations)}
    • Estimated cost accuracy: To be verified with OpenTelemetry (Phase 2)

    Generated: {json.dumps({'delegations': {json.loads(delegations_json).__len__()}})} delegations analyzed

    """ return html def _error_html(self, error_message: str) -> str: """Generate error HTML.""" return f""" Cost Dashboard

    """

    Troubleshooting

    No delegations found

    Ensure at least one Task() delegation exists in current session:

    # Query directly sqlite3 ~/.htmlgraph/htmlgraph.db \ "SELECT COUNT(*) FROM agent_events WHERE event_type='task_delegation';"

    Cost shows as 0

    Check that cost_tokens column is populated:

    # Check cost_tokens values sqlite3 ~/.htmlgraph/htmlgraph.db \ "SELECT cost_tokens FROM agent_events WHERE event_type='task_delegation' LIMIT 5;"

    If NULL, update with CIGS estimates

    Import errors

    Verify module is in Python path:

    uv run python3 -c "from htmlgraph.analytics.cost_analysis import CostAnalyzer; print(CostAnalyzer)"
